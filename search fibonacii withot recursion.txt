#include<iostream>
using namespace std;
int main()
{
    int f, f0 = 0, f1 = 1, n;
    f = f0 + f1;
    cout<<"Enter the Fibonacii Number :";
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        if(n==f)
        {
            cout<<"Fibo found";
            break;
        }
        else
            {
                f0 = f1;
                f1 = f;
                f = f0 + f1;
            }
    }
    if(f!=n)
    cout<<"not found";
}
