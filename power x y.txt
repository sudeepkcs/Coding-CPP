#include<iostream>
using namespace std;
void fun(int,int);

int main()
{
    int x,y;
    cout<<"Calculate x to the power y?"<<endl;
    cout<<"Enter x :";
    cin>>x;
    cout<<"Enter y :";
    cin>>y;
    fun(x,y);
}
void fun(int x,int y)
{   int temp =1;
    for(int i=1;i<=y;i++){
        temp = x*temp;
    }
    cout<<"Highest number is: "<< temp;
}
